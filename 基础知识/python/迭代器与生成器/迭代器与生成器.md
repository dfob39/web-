2.生成器
def countdown(n):    while n > 0:        yield n        n -= 1
# 创建生成器对象
generator = countdown(5)
# 通过迭代生成器获取值
print(next(generator))  # 输出: 5print(next(generator))  # 输出: 4print(next(generator))  # 输出: 3
# 使用 for 循环迭代生成器
for value in generator:    print(value)  # 输出: 2 1
